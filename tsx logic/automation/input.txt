import React from 'react';
import { useCurrentFrame, useVideoConfig, interpolate, Easing, spring, AbsoluteFill } from 'remotion';

export const compositionConfig = {
  id: 'Scene1_PhoneScreen',
  durationInSeconds: 15,
  fps: 30,
  width: 1920,
  height: 1080,
};

const seededRandom = (seed: number): number => {
  const x = Math.sin(seed * 9999) * 10000;
  return x - Math.floor(x);
};

const Scene1_PhoneScreen: React.FC = () => {
  const frame = useCurrentFrame();
  const { fps, durationInFrames } = useVideoConfig();

  const sp = (delay: number, config?: object) =>
    spring({ frame: Math.max(0, frame - delay), fps, config: { damping: 12, mass: 0.4, stiffness: 120, ...config } });

  const exitOp = interpolate(frame, [durationInFrames - 15, durationInFrames], [1, 0], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
  });

  // Screen crack reveal
  const crackReveal = interpolate(frame, [0, 25], [0, 1], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp', easing: Easing.out(Easing.expo),
  });

  // Danger pulse
  const dangerPulse = 1 + Math.sin(frame * 0.15) * 0.03;

  // Phone shake
  const shakeX = frame > 60 ? Math.sin(frame * 0.8) * 4 * Math.exp(-(frame - 60) * 0.03) : 0;
  const shakeY = frame > 60 ? Math.cos(frame * 1.1) * 2 * Math.exp(-(frame - 60) * 0.03) : 0;

  // Scan line moving down
  const scanLine = interpolate(frame, [20, 120], [0, 1080], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
  });

  // Word reveal
  const word1Op = sp(10);
  const word2Op = sp(25);
  const word3Op = sp(40);

  // Camera slow zoom
  const camZoom = interpolate(frame, [0, durationInFrames], [1, 1.04], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
  });

  return (
    <AbsoluteFill style={{ backgroundColor: '#1C1C1C', overflow: 'hidden', opacity: exitOp }}>
      {/* Camera zoom wrapper */}
      <div style={{ width: '100%', height: '100%', transform: `scale(${camZoom})`, transformOrigin: 'center center' }}>

        {/* Deep red-black gradient bg */}
        <div style={{
          position: 'absolute', inset: 0,
          background: 'radial-gradient(ellipse at 50% 30%, #2a0000 0%, #1C1C1C 60%, #0a0a0a 100%)',
        }} />

        {/* Animated grid — danger lines */}
        <svg style={{ position: 'absolute', inset: 0, width: '100%', height: '100%', opacity: 0.07 }}>
          {Array.from({ length: 20 }).map((_, i) => (
            <line key={`h${i}`} x1="0" y1={i * 60} x2="1920" y2={i * 60}
              stroke="#ff0000" strokeWidth="1"
              opacity={interpolate(frame, [i * 2, i * 2 + 20], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' })} />
          ))}
          {Array.from({ length: 32 }).map((_, i) => (
            <line key={`v${i}`} x1={i * 60} y1="0" x2={i * 60} y2="1080"
              stroke="#ff0000" strokeWidth="1" opacity={0.5} />
          ))}
        </svg>

        {/* Scan line moving */}
        <div style={{
          position: 'absolute', left: 0, right: 0,
          top: scanLine, height: 3,
          background: 'linear-gradient(90deg, transparent, #ff0000, transparent)',
          opacity: 0.6,
          boxShadow: '0 0 20px #ff0000',
        }} />

        {/* Giant phone silhouette */}
        <div style={{
          position: 'absolute', left: '50%', top: '50%',
          transform: `translate(-50%, -50%) translateX(${shakeX}px) translateY(${shakeY}px) scale(${dangerPulse})`,
          opacity: crackReveal,
        }}>
          <svg width="280" height="520" viewBox="0 0 280 520">
            {/* Phone body */}
            <rect x="5" y="5" width="270" height="510" rx="36" ry="36"
              fill="none" stroke="#ff0000" strokeWidth="3" opacity="0.9" />
            <rect x="15" y="15" width="250" height="490" rx="28" ry="28"
              fill="#0d0d0d" opacity="0.95" />
            {/* Screen */}
            <rect x="25" y="55" width="230" height="420" rx="8" ry="8"
              fill="#1a0000" opacity="0.8" />
            {/* Camera dot */}
            <circle cx="140" cy="35" r="8" fill="#2D2D2D" />
            <circle cx="140" cy="35" r="4" fill="#333" />
            {/* Home indicator */}
            <rect x="100" y="490" width="80" height="4" rx="2" fill="#333" />

            {/* Screen content - warning icon */}
            <text x="140" y="220" textAnchor="middle" fontSize="80" fill="#ff0000" opacity="0.9">⚠</text>
            {/* Screen glow */}
            <rect x="25" y="55" width="230" height="420" rx="8" ry="8"
              fill="none" stroke="#ff0000" strokeWidth="1" opacity={0.3 + Math.sin(frame * 0.1) * 0.2} />
          </svg>
        </div>

        {/* Eye/surveillance SVG behind phone */}
        <div style={{
          position: 'absolute', left: '50%', top: '50%',
          transform: 'translate(-50%, -50%)',
          opacity: interpolate(frame, [30, 60], [0, 0.15], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' }),
        }}>
          <svg width="600" height="300" viewBox="0 0 600 300">
            <ellipse cx="300" cy="150" rx="290" ry="140" fill="none" stroke="#ff0000" strokeWidth="2" />
            <circle cx="300" cy="150" r="80" fill="none" stroke="#ff0000" strokeWidth="2" />
            <circle cx="300" cy="150" r="40" fill="#ff0000" opacity="0.15" />
          </svg>
        </div>

        {/* Floating shapes */}
        {[
          { x: 80, y: 100, size: 120, rot: 15 },
          { x: 1700, y: 800, size: 90, rot: -20 },
          { x: 200, y: 900, size: 60, rot: 45 },
          { x: 1600, y: 150, size: 80, rot: 10 },
        ].map((s, i) => (
          <div key={i} style={{
            position: 'absolute', left: s.x, top: s.y,
            opacity: 0.05,
            transform: `rotate(${s.rot + frame * 0.02 * (i % 2 === 0 ? 1 : -1)}deg)`,
          }}>
            <svg width={s.size} height={s.size} viewBox="0 0 100 100">
              <polygon points="50,5 95,95 5,95" fill="none" stroke="#ff0000" strokeWidth="3" />
            </svg>
          </div>
        ))}

        {/* Floating particles */}
        {Array.from({ length: 18 }).map((_, i) => {
          const px = seededRandom(i * 7) * 1920;
          const py = seededRandom(i * 13) * 1080;
          const psize = 2 + seededRandom(i * 3) * 4;
          const pspeed = 0.3 + seededRandom(i * 5) * 0.5;
          return (
            <div key={i} style={{
              position: 'absolute',
              left: px,
              top: (py + frame * pspeed * 30) % 1080,
              width: psize, height: psize,
              borderRadius: '50%',
              backgroundColor: '#ff0000',
              opacity: 0.15 + seededRandom(i * 11) * 0.2,
            }} />
          );
        })}

        {/* Main text */}
        <div style={{
          position: 'absolute', left: 0, right: 0, bottom: 200,
          textAlign: 'center',
          padding: '0 120px',
        }}>
          {/* Label */}
          <div style={{
            opacity: word1Op,
            transform: `translateY(${interpolate(word1Op, [0, 1], [30, 0])}px)`,
            marginBottom: 16,
          }}>
            <span style={{
              fontFamily: 'Inter, Poppins',
              fontSize: 28,
              fontWeight: 700,
              color: '#ff0000',
              letterSpacing: 6,
              textTransform: 'uppercase',
            }}>CHAPTER 6 — DANGER ZONE</span>
          </div>

          {/* Headline */}
          <div style={{
            opacity: word2Op,
            transform: `translateY(${interpolate(word2Op, [0, 1], [40, 0])}px)`,
          }}>
            <span style={{
              fontFamily: 'Noto Sans Devanagari',
              fontSize: 100,
              fontWeight: 900,
              color: '#F5F5F5',
              lineHeight: 1.1,
              display: 'block',
            }}>सबसे खतरनाक खेल</span>
          </div>

          {/* Sub */}
          <div style={{
            opacity: word3Op,
            transform: `translateY(${interpolate(word3Op, [0, 1], [30, 0])}px)`,
            marginTop: 20,
          }}>
            <span style={{
              fontFamily: 'Noto Sans Devanagari',
              fontSize: 56,
              fontWeight: 600,
              color: '#ff0000',
            }}>आपकी मोबाइल स्क्रीन पर</span>
          </div>
        </div>

        {/* Stick figure - Buffett */}
        <div style={{
          position: 'absolute', right: 160, bottom: 80,
          opacity: sp(50),
          transform: `translateY(${interpolate(sp(50), [0, 1], [60, 0])}px)`,
        }}>
          <svg width="80" height="140" viewBox="0 0 80 140">
            {/* Golden glow */}
            <circle cx="40" cy="25" r="30" fill="#FFC107" opacity="0.08" />
            {/* Head */}
            <circle cx="40" cy="25" r="18" fill="none" stroke="#F5F5F5" strokeWidth="2.5" />
            {/* Glasses */}
            <circle cx="33" cy="25" r="6" fill="none" stroke="#FFC107" strokeWidth="2" />
            <circle cx="47" cy="25" r="6" fill="none" stroke="#FFC107" strokeWidth="2" />
            <line x1="27" y1="25" x2="20" y2="25" stroke="#FFC107" strokeWidth="2" />
            <line x1="39" y1="25" x2="41" y2="25" stroke="#FFC107" strokeWidth="2" />
            <line x1="53" y1="25" x2="60" y2="25" stroke="#FFC107" strokeWidth="2" />
            {/* Body */}
            <line x1="40" y1="43" x2="40" y2="95"
              stroke="#F5F5F5" strokeWidth="2.5"
