  width: 1920,

};

const seededRandom = (seed: number): number => {
  const x = Math.sin(seed * 9999) * 10000;
  return x - Math.floor(x);
};

    extrapolateRight: 'clamp',

  const gradColor1 = '#0D0D2B';
  const gradColor2 = '#1A0A2E';
  const gradColor3 = '#2D1B69';

  // 4% text reveal
  const fourPercentScale = sp(20, { damping: 10, stiffness: 80 });
  const fourPercentRotate = interpolate(sp(20, { damping: 10, stiffness: 80 }), [0, 1], [-15, 0]);
  const fourPercentGlow = interpolate(frame, [30, 60, 90, 120], [0, 30, 15, 30], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
  });

  // Rule text
  const ruleTextOp = interpolate(frame, [50, 70], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });
  const ruleTextY = interpolate(sp(50), [0, 1], [60, 0]);

  // Concept label
  const conceptOp = interpolate(frame, [15, 30], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });
  const conceptX = interpolate(sp(15), [0, 1], [-80, 0]);

  // Pie chart animation
  const pieProgress = interpolate(frame, [70, 140], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });
  const pieScale = sp(65);

  // Floating particles
  const particles = Array.from({ length: 25 }, (_, i) => ({
    x: seededRandom(i * 3 + 1) * 1920,
    y: seededRandom(i * 3 + 2) * 1080,
    size: seededRandom(i * 3 + 3) * 4 + 1,
    speed: seededRandom(i * 3 + 4) * 0.5 + 0.2,
    opacity: seededRandom(i * 3 + 5) * 0.3 + 0.1,
  }));

  // Shape layers
  const shapes = Array.from({ length: 6 }, (_, i) => ({
    x: seededRandom(i * 7 + 10) * 1920,
    y: seededRandom(i * 7 + 11) * 1080,
    size: seededRandom(i * 7 + 12) * 300 + 100,
    rotation: seededRandom(i * 7 + 13) * 360,
    type: i % 3,
    opacity: seededRandom(i * 7 + 14) * 0.05 + 0.03,
  }));

  // Screen shake on 4% entrance
  const shakeX = frame > 20 && frame < 35 ? Math.sin(frame * 2.5) * (35 - frame) * 0.3 : 0;
  const shakeY = frame > 20 && frame < 35 ? Math.cos(frame * 3) * (35 - frame) * 0.2 : 0;

  return (

      {/* Dynamic gradient background */}
      <AbsoluteFill

        }}
      />

      {/* Animated shape layers */}
      {shapes.map((s, i) => (
        <div
          key={`shape-${i}`}

            position: 'absolute',
            left: s.x - s.size / 2,

            height: s.size,
            opacity: s.opacity,

          }}
        >
          {s.type === 0 && (
            <div style={{ width: '100%', height: '100%', borderRadius: '50%', border: '2px solid #FFC107' }} />
          )}
          {s.type === 1 && (
            <div style={{ width: '100%', height: '100%', border: '2px solid #e9d700', borderRadius: 8 }} />
          )}
          {s.type === 2 && (
            <svg width={s.size} height={s.size} viewBox="0 0 100 100">
              <polygon points="50,5 95,95 5,95" fill="none" stroke="#F5F5F5" strokeWidth="1.5" />
            </svg>
          )}
        </div>
      ))}

      {/* Motion texture - radial grid lines */}
      <AbsoluteFill style={{ opacity: 0.04 }}>
        {Array.from({ length: 12 }, (_, i) => {
          const angle = (i * 30) + frame * 0.1;
          return (
            <div
              key={`line-${i}`}
              style={{
                position: 'absolute',
                left: 960,
                top: 540,
                width: 1400,

                transformOrigin: '0 50%',

              }}
            />

            gap: 15,

        >

            style={{
              width: 8,
              height: 40,
              backgroundColor: '#FFC107',
              borderRadius: 4,

          />
          <span
            style={{
              fontFamily: 'Noto Sans Devanagari',
              fontSize: 36,
              fontWeight: 600,
              color: '#9E9E9E',
              letterSpacing: 3,
            }}
          >
            ज़रूरी CONCEPT
          </span>
        </div>

        {/* Giant 4% text center */}
        <div
          style={{
            position: 'absolute',
            top: '50%',
            left: '50%',
            transform: `translate(-50%, -55%) scale(${fourPercentScale}) rotate(${fourPercentRotate}deg)`,
          }}
        >
          <div
            style={{
              fontFamily: 'Inter',
              fontSize: 260,
              fontWeight: 900,
              color: '#e9d700',
              textShadow: `0 0 ${fourPercentGlow}px #e9d700, 0 0 ${fourPercentGlow * 2}px rgba(233,215,0,0.3)`,
              lineHeight: 1,

            4%

              </div>
              {/* Mail icon bounce */}
              <svg
                style={{ position: 'absolute', right: 30, bottom: 30, transform: `translateY(${mailBounce}px)` }}
                width="40"
                height="40"
                viewBox="0 0 40 40"
              >
                <rect x="4" y="8" width="32" height="24" rx="4" fill="none" stroke="#FFC107" strokeWidth="2" />
