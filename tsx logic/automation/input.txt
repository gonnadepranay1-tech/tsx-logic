  });

  // "CHANGE" word impact
  const changeImpact = sp(75, { damping: 6, stiffness: 250, mass: 0.5 });

  // Light burst on break
  const lightBurst = interpolate(frame, [70, 88], [0, 1], {
    extrapolateLeft: 'clamp',
    extrapolateRight: 'clamp',
  });

  const particles = Array.from({ length: 20 }, (_, i) => ({
    x: seededRandom(i * 4 + 30) * 1920,
    y: seededRandom(i * 4 + 31) * 1080,
    r: seededRandom(i * 4 + 32) * 4 + 1.5,
    speed: seededRandom(i * 6 + 6) * 0.45 + 0.1,
    op: seededRandom(i * 5 + 10) * 0.2 + 0.05,
  }));

  return (
    <AbsoluteFill
      style={{
        background: `linear-gradient(150deg, #080808 0%, ${
          breakProgress > 0.3 ? `rgba(30,18,0,${breakProgress * 0.8 + 0.2})` : '#141414'
        } 50%, #080808 100%)`,
        overflow: 'hidden',
        opacity: exitOp,
      }}
    >
      {/* Dynamic light burst */}
      <AbsoluteFill
        style={{
          background: `radial-gradient(ellipse 60% 50% at 50% 50%, rgba(233,215,0,${lightBurst * 0.12}) 0%, transparent 65%)`,
        }}
      />

      {/* === SHAPE LAYERS — Prison cell aesthetic === */}
      {/* Corner shadows — dark */}
      <div style={{ position: 'absolute', top: 0, left: 0, width: 300, height: '100%', background: 'linear-gradient(to right, rgba(0,0,0,0.5), transparent)' }} />
      <div style={{ position: 'absolute', top: 0, right: 0, width: 300, height: '100%', background: 'linear-gradient(to left, rgba(0,0,0,0.5), transparent)' }} />

      {/* === MOTION TEXTURE — horizontal cell lines === */}
      <svg style={{ position: 'absolute', inset: 0, opacity: 0.02 }} width={1920} height={1080}>
        {Array.from({ length: 18 }, (_, i) => (
          <line
            key={i}
            x1={0}
            y1={i * 60}
            x2={1920}
            y2={i * 60}
            stroke="#ffffff"
            strokeWidth={1}
          />
        ))}
      </svg>

      {/* === PARTICLES — dust/debris from breaking === */}
      <svg style={{ position: 'absolute', inset: 0 }} width={1920} height={1080}>
        {particles.map((p, i) => (
          <circle
            key={i}
            cx={
              960 +
              (p.x - 960) * (1 + breakProgress * 0.3) +
              Math.sin(frame * p.speed + i) * 12
            }
            cy={
              540 +
              (p.y - 540) * (1 + breakProgress * 0.2) +
              Math.cos(frame * p.speed * 0.7 + i) * 8
            }
            r={p.r * (1 + breakProgress)}
            fill={i % 3 === 0 ? '#e9d700' : i % 3 === 1 ? '#FFC107' : '#9E9E9E'}
            opacity={p.op * (1 - breakProgress * 0.3)}
          />
        ))}
      </svg>

      {/* === PRISON BARS — center === */}
      <div
        style={{
          position: 'absolute',
          left: '50%',
          top: '10%',
          transform: `translateX(-50%) scale(${barsReveal})`,
          opacity: barsReveal,
        }}
      >
        <svg width={700} height={780} viewBox="0 0 700 780">
          {/* Top bar */}
          <rect
            x={20}
            y={20}
            width={660}
            height={20}
            rx={4}
            fill="rgba(100,100,100,0.5)"
            stroke="#9E9E9E"
            strokeWidth={1.5}
          />
          {/* Bottom bar */}
          <rect
            x={20}
            y={740}
            width={660}
            height={20}
            rx={4}
            fill="rgba(100,100,100,0.5)"
            stroke="#9E9E9E"
            strokeWidth={1.5}
          />

          {/* Vertical bars — outer ones stay, middle 3 break */}
          {[0, 1, 2, 3, 4, 5, 6].map((i) => {
            const x = 50 + i * 100;
            const isMiddle = i >= 2 && i <= 4;
            const breakOffset = isMiddle
              ? (i - 3) * breakProgress * 120
              : 0;
            const breakAngle = isMiddle
              ? (i - 3) * breakProgress * 25
              : 0;
            const barOpacity = isMiddle
              ? 1 - breakProgress * 0.8
              : 1;

            return (
              <rect
                key={i}
                x={x + breakOffset}
                y={40}
                width={24}
                height={700}
                rx={5}
                fill={`rgba(80,80,80,${barOpacity * 0.6})`}
                stroke={`rgba(160,160,160,${barOpacity})`}
                strokeWidth={1.5}
                transform={`rotate(${breakAngle}, ${x + 12}, 390)`}
              />
            );
          })}

          {/* Stick figure trapped INSIDE */}
          <g
            transform={`translate(350, 390) scale(${0.7 + breakProgress * 0.3})`}
            opacity={barsReveal}
          >
            <circle cx={0} cy={-120} r={22} fill="none" stroke="#F5F5F5" strokeWidth={2} />
            {/* Glasses */}
            <circle cx={-8} cy={-120} r={8} fill="none" stroke="#e9d700" strokeWidth={1.5} />
            <circle cx={11} cy={-120} r={8} fill="none" stroke="#e9d700" strokeWidth={1.5} />
            <line x1={3} y1={-120} x2={3} y2={-120} stroke="#e9d700" strokeWidth={1} />
            {/* Expression shifts from sad to hopeful */}
            <path
              d={
                breakProgress > 0.5
                  ? `M-8,-113 Q0,-108 8,-113`
                  : `M-8,-113 Q0,-118 8,-113`
              }
              fill="none"
              stroke="#F5F5F5"
              strokeWidth={1.5}
            />
            <line x1={0} y1={-98} x2={0} y2={-48} stroke="#F5F5F5" strokeWidth={2} />
            <line x1={0} y1={-80} x2={-22} y2={-68} stroke="#F5F5F5" strokeWidth={1.8} />
            <line x1={0} y1={-80} x2={22} y2={-68} stroke="#F5F5F5" strokeWidth={1.8} />
            <line x1={0} y1={-48} x2={-14} y2={0} stroke="#F5F5F5" strokeWidth={1.8} />
            <line x1={0} y1={-48} x2={14} y2={0} stroke="#F5F5F5" strokeWidth={1.8} />
          </g>

          {/* LABEL on bars */}
          <text
            x={350}
            y={400}
            textAnchor="middle"
            fill={`rgba(255,68,68,${1 - breakProgress})`}
            fontSize={28}
            fontFamily="Inter, Poppins"
            fontWeight="bold"
            letterSpacing={8}
          >
            सावधानी की जेल
          </text>
        </svg>
      </div>

      {/* === BREAKING FREE TEXT === */}
      <div
        style={{
          position: 'absolute',
          bottom: 80,
          left: 0,
          right: 0,
          textAlign: 'center',
          transform: `scale(${changeImpact})`,
          opacity: changeImpact,
        }}
      >
        <div
          style={{
            fontFamily: 'Noto Sans Devanagari',
            fontSize: 36,
            color: '#9E9E9E',
            letterSpacing: 4,
            marginBottom: 10,
          }}
        >
          अब समय है
        </div>
        <div
          style={{
            fontFamily: 'Inter, Poppins',
            fontSize: 110,
            fontWeight: 900,
            color: '#e9d700',
            letterSpacing: -2,
            lineHeight: 1,
            textShadow: `0 0 ${lightBurst * 60}px rgba(233,215,0,${lightBurst * 0.5})`,
          }}
        >
          बदलाव का
        </div>
      </div>

      {/* === VIGNETTE === */}
      <AbsoluteFill
        style={{
          background:
            'radial-gradient(ellipse 80% 75% at 50% 50%, transparent 40%, rgba(0,0,0,0.9) 100%)',
          pointerEvents: 'none',
        }}
      />

      {/* === WATERMARK === */}
      <div
        style={{
          position: 'absolute',
          bottom: 30,
          right: 45,
          fontFamily: 'Inter, Poppins',
          fontSize: 22,
          fontWeight: 600,
          color: '#9E9E9E',
          opacity: 0.3,
          letterSpacing: 1,
        }}
      >
        ProBuffet HQ
      </div>
    </AbsoluteFill>
  );
};

export default C6S10_PrisonOfCaution;
