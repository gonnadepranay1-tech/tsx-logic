export const compositionConfig = {
  id: 'C6S1_TinHisse',
  durationInSeconds: 12,
  fps: 30,
  width: 1920,
  height: 1080,
};

const seededRandom = (seed: number): number => {
  const x = Math.sin(seed * 9999) * 10000;
  return x - Math.floor(x);
};

  const frame = useCurrentFrame();
  const { fps, durationInFrames } = useVideoConfig();

    spring({ frame: Math.max(0, frame - delay), fps, config: { damping: 12, mass: 0.4, stiffness: 120, ...config } });

  const exitOp = interpolate(frame, [durationInFrames - 15, durationInFrames], [1, 0], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
  });

  // Coin split animation
  const coinScale = sp(0, { damping: 8, stiffness: 80 });
  const splitLeft = sp(20);
  const splitMid = sp(35);
  const splitRight = sp(50);

  const titleY = interpolate(sp(5), [0, 1], [80, 0]);
  const titleOp = interpolate(sp(5), [0, 1], [0, 1]);

  // Shake on "बांटो"
  const shakeX = frame > 55 && frame < 65
    ? Math.sin(frame * 1.8) * 6
    : 0;

  // Background pulse
  const bgPulse = interpolate(frame, [0, 60, 120, 180, 360], [0, 0.06, 0.03, 0.06, 0.03], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
  });

  return (
    <AbsoluteFill style={{ backgroundColor: '#1C1C1C', overflow: 'hidden', opacity: exitOp }}>

      {/* Dynamic gradient background */}
      <AbsoluteFill style={{
        background: `radial-gradient(ellipse at 20% 50%, rgba(233,215,0,${0.08 + bgPulse}) 0%, transparent 55%),
                     radial-gradient(ellipse at 80% 30%, rgba(255,193,7,0.06) 0%, transparent 50%),
                     radial-gradient(ellipse at 50% 90%, rgba(45,45,45,0.9) 0%, transparent 60%)`,
      }} />

      {/* Shape layer 1 — large rotating hexagon */}
      <svg style={{ position: 'absolute', top: -100, left: -80, opacity: 0.04,
        transform: `rotate(${frame * 0.15}deg)` }} width={600} height={600}>

          fill="none" stroke="#e9d700" strokeWidth={3} />
      </svg>

      {/* Shape layer 2 — circle drift */}
      <div style={{
        position: 'absolute', top: 120, right: -60,
        width: 400, height: 400, borderRadius: '50%',
        border: '2px solid rgba(255,193,7,0.05)',
        transform: `translateY(${Math.sin(frame * 0.02) * 20}px) scale(${1 + Math.sin(frame * 0.03) * 0.02})`,
      }} />

      {/* Motion texture — horizontal split lines */}
      <svg style={{ position: 'absolute', inset: 0, opacity: 0.035 }} width={1920} height={1080}>
        {[...Array(5)].map((_, i) => (
          <line key={i} x1={0} y1={200 + i * 160} x2={1920} y2={200 + i * 160}
            stroke="#9E9E9E" strokeWidth={1}
            strokeDasharray={`${40 + i * 12},${80 + i * 8}`}
            strokeDashoffset={frame * (i % 2 === 0 ? -1.2 : 1.2)} />
        ))}
      </svg>

      {/* Floating particles */}
      {[...Array(18)].map((_, i) => {
        const px = seededRandom(i * 13) * 1920;
        const py = seededRandom(i * 17) * 1080;
        const pSize = seededRandom(i * 7) * 4 + 2;

            position: 'absolute',

        );

        position: 'absolute', inset: 0,

        alignItems: 'center', justifyContent: 'center',
        transform: `translateX(${shakeX}px)`,

            दूसरा कदम
