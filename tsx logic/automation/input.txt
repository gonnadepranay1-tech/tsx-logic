import React from 'react';

  width: 1920,
  height: 1080,

  const x = Math.sin(seed * 9999) * 10000;
  return x - Math.floor(x);
};

  const frame = useCurrentFrame();
  const { fps, durationInFrames } = useVideoConfig();

      frame: Math.max(0, frame - delay),
      fps,
      config: { damping: 12, mass: 0.4, stiffness: 120, ...cfg },
    });

  const exitOp = interpolate(frame, [durationInFrames - 15, durationInFrames], [1, 0], {
    extrapolateLeft: 'clamp',
    extrapolateRight: 'clamp',
  });

  // Zoom out slow
  const zoom = interpolate(frame, [0, durationInFrames], [1.05, 1.0], {
    extrapolateLeft: 'clamp',
    extrapolateRight: 'clamp',
  });

  // Graph line progress
  const lineProgress = interpolate(frame, [15, 80], [0, 1], {
    extrapolateLeft: 'clamp',
    extrapolateRight: 'clamp',
    easing: Easing.out(Easing.cubic),
  });

  const graphW = 900;
  const graphH = 380;
  const points = Array.from({ length: 21 }, (_, i) => {
    const t = i / 20;

    extrapolateLeft: 'clamp',

    x: seededRandom(i * 5) * 1920,

    speed: seededRandom(i * 5 + 4) * 0.015 + 0.005,
  }));

  return (
    <AbsoluteFill
      style={{
        background: 'linear-gradient(160deg, #020810 0%, #051220 50%, #0A1A10 100%)',
        overflow: 'hidden',
        opacity: exitOp,

          position: 'absolute',
          inset: 0,
          transform: `scale(${zoom})`,

          {Array.from({ length: 40 }, (_, i) => (
            <line
              key={i}
              x1={i * 50 - 500}
              y1={0}
              x2={i * 50 + 500}
              y2={1080}
              stroke="#00FF88"
              strokeWidth={1}
            />
          ))}
        </svg>

            const s = h.size;
            const pts = Array.from({ length: 6 }, (_, j) => {
              const a = (j * 60 - 30) * (Math.PI / 180);
              return `${cx + s * Math.cos(a)},${cy + s * Math.sin(a)}`;

              width: seededRandom(i * 4 + 2) * 4 + 1.5,
              height: seededRandom(i * 4 + 2) * 4 + 1.5,
              borderRadius: '50%',
              background: i % 4 === 0 ? '#e9d700' : '#00FF88',
              opacity: 0.1 + Math.sin(frame * 0.06 + i) * 0.05,
            }}
          />
        ))}
