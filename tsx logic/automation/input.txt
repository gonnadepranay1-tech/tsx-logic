  id: 'C6S1_JetztWirdEsKonkret',
  durationInSeconds: 12,
  fps: 30,
  width: 1920,
  height: 1080,
};

  const x = Math.sin(seed * 9999) * 10000;
  return x - Math.floor(x);
};

    frame,
    [durationInFrames - 15, durationInFrames],
    [1, 0],
    { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' }

    extrapolateRight: 'clamp',

  const scanX = interpolate(frame, [0, durationInFrames], [-200, 2200], {
    extrapolateLeft: 'clamp',
    extrapolateRight: 'clamp',
  });

  const headNod = Math.sin(frame * 0.04) * 3;
  const figureScale = sp(40);

  return (
    <AbsoluteFill
      style={{
        backgroundColor: '#1C1C1C',
        overflow: 'hidden',
        opacity: exitOp,
        transform: `scale(${camScale})`,

          background:
            'radial-gradient(ellipse at 20% 50%, #2a2200 0%, #1C1C1C 55%), radial-gradient(ellipse at 80% 50%, #1a1a00 0%, transparent 60%)',
        }}
      />

      <AbsoluteFill style={{ opacity: 1 }}>
        {/* Large rotating rectangle */}
        <div
          style={{
            position: 'absolute',
            width: 700,
            height: 700,
            border: '2px solid #e9d700',
            opacity: 0.04,
            top: -150,
            left: -100,
            transform: `rotate(${frame * 0.15}deg)`,
            borderRadius: 8,
          }}
        />
        {/* Medium circle drifting */}
        <div
          style={{
            position: 'absolute',
            width: 500,
            height: 500,
            border: '1.5px solid #F5F5F5',
            opacity: 0.03,
            bottom: -100,
            right: 100,
            borderRadius: '50%',
            transform: `scale(${1 + Math.sin(frame * 0.02) * 0.05})`,
          }}
        />
        {/* Triangle SVG */}
        <svg
          style={{
            position: 'absolute',
            top: 200,
            right: 300,
            opacity: 0.05,
            transform: `rotate(${frame * 0.08}deg)`,
          }}
          width="300"
          height="300"
          viewBox="0 0 300 300"
        >
          <polygon points="150,10 290,290 10,290" fill="none" stroke="#e9d700" strokeWidth="2" />
        </svg>
      </AbsoluteFill>

      {/* Motion texture — horizontal scan lines */}
      <AbsoluteFill style={{ opacity: 1 }}>
        {Array.from({ length: 18 }).map((_, i) => (
          <div
            key={i}

              position: 'absolute',
              left: 0,

              key={i}
              style={{
                position: 'absolute',
                left: px,
