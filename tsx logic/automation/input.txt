import React from 'react';
import { useCurrentFrame, useVideoConfig, interpolate, Easing, spring, AbsoluteFill } from 'remotion';

export const compositionConfig = {
  id: 'PensionOpener',
  durationInSeconds: 12,
  fps: 30,
  width: 1920,
  height: 1080,
};

const seededRandom = (seed: number): number => {
  const x = Math.sin(seed * 9999) * 10000;
  return x - Math.floor(x);
};

const PensionOpener: React.FC = () => {
  const frame = useCurrentFrame();
  const { fps, durationInFrames } = useVideoConfig();

  const sp = (delay: number, config?: object) =>
    spring({ frame: Math.max(0, frame - delay), fps, config: { damping: 12, mass: 0.4, stiffness: 120, ...config } });

  const exitOp = interpolate(frame, [durationInFrames - 15, durationInFrames], [1, 0], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
  });

  // Slow time ticker scanning across screen
  const scanX = interpolate(frame, [0, durationInFrames], [-200, 2200], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
  });

  // Pulsing background glow
  const glowPulse = Math.sin(frame * 0.06) * 0.5 + 0.5;

  // Title split reveal
  const line1Op = interpolate(frame, [10, 30], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });
  const line2Op = interpolate(frame, [30, 50], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });
  const line1Y = sp(10) * 0 + interpolate(frame, [10, 35], [60, 0], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp', easing: Easing.out(Easing.cubic) });
  const line2Y = interpolate(frame, [30, 55], [60, 0], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp', easing: Easing.out(Easing.cubic) });

  // Chapter badge
  const badgeScale = sp(5, { stiffness: 200, damping: 14 });

  // Clock hands animation
  const secondHand = (frame / fps) * 360;
  const minuteHand = (frame / fps / 60) * 360;

  return (
    <AbsoluteFill style={{ backgroundColor: '#1C1C1C', overflow: 'hidden', opacity: exitOp }}>
      {/* Deep radial gradient — dark burgundy core */}
      <AbsoluteFill>
        <div style={{
          position: 'absolute', inset: 0,
          background: `radial-gradient(ellipse 70% 60% at 50% 50%, #2a0a0a ${glowPulse * 10}%, #1C1C1C 70%)`,
        }} />
      </AbsoluteFill>

      {/* Horizontal scan line */}
      <div style={{
        position: 'absolute', top: 0, bottom: 0,
        left: scanX, width: 3,
        background: 'linear-gradient(to bottom, transparent, #e9d700 20%, #e9d70088 80%, transparent)',
        boxShadow: '0 0 30px #e9d700aa',
      }} />

      {/* Grid texture — fine horizontal lines */}
      <AbsoluteFill>
        <svg width="1920" height="1080" style={{ opacity: 0.04 }}>
          {Array.from({ length: 54 }).map((_, i) => (
            <line key={i} x1="0" y1={i * 20} x2="1920" y2={i * 20} stroke="#F5F5F5" strokeWidth="1" />
          ))}
          {Array.from({ length: 97 }).map((_, i) => (
            <line key={`v${i}`} x1={i * 20} y1="0" x2={i * 20} y2="1080" stroke="#F5F5F5" strokeWidth="0.5" />
          ))}
        </svg>
      </AbsoluteFill>

      {/* Floating geometric shapes */}
      <AbsoluteFill style={{ opacity: 0.06 }}>
        {/* Large slow drifting hexagon */}
        <svg width="1920" height="1080" style={{ position: 'absolute' }}>
          <polygon
            points="960,80 1120,170 1120,350 960,440 800,350 800,170"
            fill="none" stroke="#e9d700" strokeWidth="2"
            transform={`translate(${Math.sin(frame * 0.008) * 15}, ${Math.cos(frame * 0.006) * 10}) rotate(${frame * 0.04}, 960, 260)`}
          />
          <circle cx="200" cy="800" r="180"
            fill="none" stroke="#F5F5F5" strokeWidth="1.5"
            transform={`scale(${1 + Math.sin(frame * 0.05) * 0.04})`}
            style={{ transformOrigin: '200px 800px' }}
          />
          <rect x="1600" y="100" width="200" height="200"
            fill="none" stroke="#F5F5F5" strokeWidth="1"
            transform={`rotate(${45 + frame * 0.03}, 1700, 200)`}
          />
        </svg>
      </AbsoluteFill>

      {/* Floating particles — golden dust */}
      <AbsoluteFill>
        {Array.from({ length: 22 }).map((_, i) => {
          const seed = i * 137;
          const x = seededRandom(seed) * 1920;
          const startY = seededRandom(seed + 1) * 1080;
          const speed = seededRandom(seed + 2) * 0.4 + 0.15;
          const y = ((startY + frame * speed) % 1080);
          const size = seededRandom(seed + 3) * 3 + 1;

          return (
            <div key={i} style={{
              position: 'absolute', left: x, top: y,

              <line key={i}

                x2={140 + Math.cos(rad) * 118} y2={140 + Math.sin(rad) * 118}
                stroke="#e9d700" strokeWidth="3" />
            );
          })}
          {/* Minute hand */}
          <line
            x1="140" y1="140"
            x2={140 + Math.cos((minuteHand - 90) * Math.PI / 180) * 80}
            y2={140 + Math.sin((minuteHand - 90) * Math.PI / 180) * 80}
            stroke="#F5F5F5" strokeWidth="4" strokeLinecap="round"
