import React from 'react';
import {
  useCurrentFrame,
  useVideoConfig,
  interpolate,
  spring,
  AbsoluteFill,
} from 'remotion';

export const compositionConfig = {
  id: 'Ch6Scene1',
  durationInSeconds: 12,
  fps: 30,
  width: 1920,
  height: 1080,
};

const seededRandom = (seed: number): number => {
  const x = Math.sin(seed * 9999) * 10000;
  return x - Math.floor(x);
};

const Ch6Scene1: React.FC = () => {
  const frame = useCurrentFrame();
  const { fps, durationInFrames } = useVideoConfig();

  const sp = (delay: number, config?: object) =>
    spring({
      frame: Math.max(0, frame - delay),
      fps,
      config: { damping: 12, mass: 0.4, stiffness: 120, ...config },
    });

  const exitOp = interpolate(
    frame,
    [durationInFrames - 15, durationInFrames],
    [1, 0],
    { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' }
  );

  // Cage bars reveal
  const cageOp = interpolate(frame, [0, 20], [0, 1], {
    extrapolateLeft: 'clamp',
    extrapolateRight: 'clamp',
  });

  // Crack animation
  const crackScale = sp(40, { damping: 8, stiffness: 200 });
  const crackOp = interpolate(frame, [40, 55], [0, 1], {
    extrapolateLeft: 'clamp',
    extrapolateRight: 'clamp',
  });

  const headlineY = interpolate(sp(10), [0, 1], [80, 0]);
  const subY = interpolate(sp(25), [0, 1], [60, 0]);
  const wordOp = interpolate(sp(50), [0, 1], [0, 1]);

  // Shake effect on crack
  const shakeX =
    frame > 42 && frame < 52
      ? Math.sin(frame * 0.8) * 6 * (1 - (frame - 42) / 10)
      : 0;

  // Floating particles
  const particles = Array.from({ length: 18 }, (_, i) => ({
    x: seededRandom(i * 7) * 1920,
    y: seededRandom(i * 13) * 1080,
    r: seededRandom(i * 3) * 4 + 2,
    speed: seededRandom(i * 17) * 0.4 + 0.1,
    phase: seededRandom(i * 11) * Math.PI * 2,
  }));

  // Background grid lines
  const gridLines = Array.from({ length: 8 }, (_, i) => i);

  return (
    <AbsoluteFill
      style={{
        backgroundColor: '#1C1C1C',
        overflow: 'hidden',
        opacity: exitOp,
        transform: `translateX(${shakeX}px)`,
      }}
    >
      {/* Gradient background — deep crimson to black */}
      <AbsoluteFill
        style={{
          background:
            'radial-gradient(ellipse at 30% 50%, #3a0a0a 0%, #1C1C1C 65%)',
        }}
      />

      {/* Grid texture */}
      <AbsoluteFill style={{ opacity: 0.04 }}>
        <svg width="1920" height="1080">
          {gridLines.map((i) => (
            <React.Fragment key={i}>
              <line
                x1={i * 240}
                y1={0}
                x2={i * 240}
                y2={1080}
                stroke="#F5F5F5"
                strokeWidth="1"
              />
              <line
                x1={0}
                y1={i * 135}
                x2={1920}
                y2={i * 135}
                stroke="#F5F5F5"
                strokeWidth="1"
              />
            </React.Fragment>
          ))}
        </svg>
      </AbsoluteFill>

      {/* Shape layers — large geometric drifters */}
      <AbsoluteFill style={{ opacity: 0.05 }}>
        <svg width="1920" height="1080">
          {/* Large circle top-right */}
          <circle
            cx={1600 + Math.sin(frame * 0.008) * 15}
            cy={200 + Math.cos(frame * 0.006) * 12}
            r={320}
            fill="none"
            stroke="#ff0000"
            strokeWidth="2"
            transform={`rotate(${frame * 0.03}, 1600, 200)`}
          />
          {/* Rectangle bottom-left */}
          <rect
            x={-80 + Math.sin(frame * 0.007) * 20}
            y={700 + Math.cos(frame * 0.009) * 15}
            width={400}
            height={400}
            fill="none"
            stroke="#F5F5F5"
            strokeWidth="1.5"
            transform={`rotate(${frame * 0.04}, 120, 900)`}
          />
          {/* Triangle center-right */}
          <polygon
            points={`1400,${300 + Math.sin(frame * 0.01) * 10} 1550,${560 + Math.cos(frame * 0.008) * 10} 1250,${560 + Math.sin(frame * 0.009) * 10}`}
            fill="none"
            stroke="#e9d700"
            strokeWidth="1"
          />
        </svg>
      </AbsoluteFill>

      {/* Cage bars — vertical lines */}
      <AbsoluteFill style={{ opacity: cageOp * 0.18 }}>
        <svg width="1920" height="1080">
          {[0, 1, 2, 3, 4, 5, 6, 7].map((i) => (
            <rect
              key={i}
              x={160 + i * 200}
              y={0}
              width={28}
              height={1080}
              fill="#ff0000"
              rx={4}
            />
          ))}
          {/* Horizontal bars */}
          <rect x={160} y={100} width={1440} height={20} fill="#ff0000" rx={4} />
          <rect x={160} y={960} width={1440} height={20} fill="#ff0000" rx={4} />
        </svg>
      </AbsoluteFill>

      {/* CRACK SVG on the cage — center reveal */}
      <AbsoluteFill
        style={{ opacity: crackOp, transform: `scale(${0.8 + crackScale * 0.2})` }}

        <svg width="1920" height="1080">

            stroke="#e9d700"
            strokeWidth="4"
            fill="none"
            strokeLinecap="round"

            stroke="#e9d700"
            strokeWidth="3"
            fill="none"
            strokeLinecap="round"
          />
          {/* Glow around crack */}

            cx={968}
            cy={545}
            rx={60}
            ry={80}
            fill="#e9d700"
            opacity={0.06}
          />
        </svg>
      </AbsoluteFill>

      {/* Floating particles */}
      <AbsoluteFill style={{ pointerEvents: 'none' }}>
        <svg width="1920" height="1080">
          {particles.map((p, i) => (
            <circle
              key={i}
              cx={p.x + Math.sin(frame * p.speed + p.phase) * 25}
              cy={p.y - (frame * p.speed * 0.5) % 1080}
              r={p.r}
              fill={i % 3 === 0 ? '#e9d700' : i % 3 === 1 ? '#ff0000' : '#F5F5F5'}
              opacity={0.25}
            />
          ))}
        </svg>
      </AbsoluteFill>

      {/* LEFT side — large label */}
      <div
        style={{
          position: 'absolute',
          left: 120,
          top: 80,
          transform: `translateY(${headlineY}px)`,
          opacity: sp(10),
        }}
      >
        <div
          style={{
            fontFamily: 'Anton, sans-serif',
            fontSize: 22,
            color: '#ff0000',
            letterSpacing: 8,
            textTransform: 'uppercase',
            marginBottom: 16,
          }}
        >
          KAPITEL 6
        </div>
        <div
          style={{
            fontFamily: 'Anton, sans-serif',
            fontSize: 100,
            color: '#F5F5F5',
            lineHeight: 1,
            textTransform: 'uppercase',
            maxWidth: 700,
          }}
        >
          JETZT
          <br />
          WIRD ES
          <br />
          <span style={{ color: '#e9d700' }}>KONKRET</span>
        </div>
      </div>

      {/* BOTTOM — subhead */}

          </span>
